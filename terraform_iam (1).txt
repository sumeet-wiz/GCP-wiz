# iam.tf - Identity and Access Management with Misconfigurations

# Overprivileged service account (MISCONFIGURATION)
resource "google_service_account" "overprivileged_sa" {
  account_id   = "csmp-test-overprivileged-sa"
  display_name = "CSMP Test Overprivileged Service Account"
  description  = "Service account with excessive permissions for testing"
}

# Grant excessive permissions (MISCONFIGURATION)
resource "google_project_iam_member" "sa_editor" {
  project = var.project_id
  role    = "roles/editor"
  member  = "serviceAccount:${google_service_account.overprivileged_sa.email}"
}

resource "google_project_iam_member" "sa_owner" {
  project = var.project_id
  role    = "roles/owner"
  member  = "serviceAccount:${google_service_account.overprivileged_sa.email}"
}

# Service account key (MISCONFIGURATION - keys should be avoided)
resource "google_service_account_key" "vulnerable_key" {
  service_account_id = google_service_account.overprivileged_sa.name
  public_key_type    = "TYPE_X509_PEM_FILE"
}

# Create a user account with excessive permissions (if email provided)
variable "test_user_email" {
  description = "Test user email for IAM testing (optional)"
  type        = string
  default     = ""
}

resource "google_project_iam_member" "test_user_owner" {
  count   = var.test_user_email != "" ? 1 : 0
  project = var.project_id
  role    = "roles/owner"
  member  = "user:${var.test_user_email}"
}
